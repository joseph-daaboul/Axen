src/protectedrouter.jsx
import { useState, useEffect } from "react";
import { auth } from "../firebase/firebase";
import { Navigate } from "react-router-dom";
import { onAuthStateChanged } from "firebase/auth";

export default function ProtectedRoute({ children }) {
  const [loading, setLoading] = useState(true);
  const [user, setUser] = useState(null);

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      setLoading(false);
    });

    return () => unsubscribe();
  }, []);

  if (loading) {
    return (
      <div className="flex justify-center items-center h-screen">
        <p>Loading...</p>
      </div>
    );
  }

  if (!user) {
    return <Navigate to="/" />;
  }

  return children;
}

import { useState } from "react";
import Navbar from "../components/Navbar";
import { sendMessageToAI } from "../services/chatbotService"; // Make sure this matches your actual filename

export default function Chatbot() {
  const [message, setMessage] = useState("");
  const [chat, setChat] = useState([]);
  const [loading, setLoading] = useState(false);

  const sendMessage = async () => {
    if (!message.trim()) return;

    const userMsg = { sender: "user", text: message };
    setChat([...chat, userMsg]);
    setMessage("");
    setLoading(true);

    try {
      const reply = await sendMessageToAI(message);
      const aiMsg = { sender: "ai", text: reply };
      setChat((prev) => [...prev, aiMsg]);
    } catch (error) {
      console.error("Chat error:", error);
      const errorMsg = {
        sender: "ai",
        text: "Sorry, there was an error. Please try again.",
      };
      setChat((prev) => [...prev, errorMsg]);
    } finally {
      setLoading(false);
    }
  };

  const handleKeyPress = (e) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      sendMessage();
    }
  };

  return (
    <>
      <Navbar />

      <div className="p-6 max-w-2xl mx-auto">
        <h2 className="text-xl mb-4">AI Mental Assistant</h2>

        <div className="border h-96 overflow-y-auto p-4 mb-4 bg-gray-50">
          {chat.map((msg, index) => (
            <div
              key={index}
              className={`mb-3 ${msg.sender === "user" ? "text-right" : "text-left"}`}
            >
              <div
                className={`inline-block p-3 rounded-lg max-w-[80%] ${
                  msg.sender === "user"
                    ? "bg-black text-white"
                    : "bg-gray-200 text-black"
                }`}
              >
                <p className="text-sm font-semibold mb-1">
                  {msg.sender === "user" ? "You" : "AI Assistant"}
                </p>
                <p>{msg.text}</p>
              </div>
            </div>
          ))}
          {loading && (
            <div className="text-left">
              <div className="inline-block p-3 rounded-lg bg-gray-200">
                <p className="text-sm">AI is typing...</p>
              </div>
            </div>
          )}
        </div>

        <div className="flex gap-2">
          <input
            className="border w-full p-2 rounded"
            value={message}
            onChange={(e) => setMessage(e.target.value)}
            onKeyPress={handleKeyPress}
            placeholder="Ask something..."
            disabled={loading}
          />
          <button
            onClick={sendMessage}
            className="bg-black text-white px-6 rounded disabled:bg-gray-400"
            disabled={loading}
          >
            Send
          </button>
        </div>
      </div>
    </>
  );
}

import { useState, useEffect } from "react";
import Navbar from "../components/Navbar";
import { auth, db } from "../firebase/firebase";
import { doc, setDoc, getDoc, runTransaction } from "firebase/firestore";

export default function Motivation() {
  const [goals, setGoals] = useState([""]);
  const [strengths, setStrengths] = useState([""]);
  const [savedData, setSavedData] = useState({ goals: [], strengths: [] });
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);

  useEffect(() => {
    loadMotivation();
  }, []);

  const loadMotivation = async () => {
    try {
      if (!auth.currentUser) {
        setLoading(false);
        return;
      }

      const userRef = doc(db, "users", auth.currentUser.uid);
      const docSnap = await getDoc(userRef);

      if (docSnap.exists() && docSnap.data().motivation) {
        const data = docSnap.data().motivation;
        setSavedData(data);
        setGoals(data.goals.length > 0 ? [...data.goals, ""] : [""]);
        setStrengths(data.strengths.length > 0 ? [...data.strengths, ""] : [""]);
      }
    } catch (error) {
      console.error("Error loading motivation:", error);
    } finally {
      setLoading(false);
    }
  };

  const handleGoalChange = (index, value) => {
    const newGoals = [...goals];
    newGoals[index] = value;

    if (index === goals.length - 1 && value.trim() !== "") {
      newGoals.push("");
    }

    setGoals(newGoals);
  };

  const handleStrengthChange = (index, value) => {
    const newStrengths = [...strengths];
    newStrengths[index] = value;

    if (index === strengths.length - 1 && value.trim() !== "") {
      newStrengths.push("");
    }

    setStrengths(newStrengths);
  };

  const saveMotivation = async () => {
    setSaving(true);

    try {
      if (!auth.currentUser) {
        alert("Please log in again");
        return;
      }

      const filteredGoals = goals.filter((g) => g.trim() !== "");
      const filteredStrengths = strengths.filter((s) => s.trim() !== "");

      if (filteredGoals.length === 0 && filteredStrengths.length === 0) {
        alert("Please enter at least one goal or strength");
        return;
      }

      const motivationData = {
        goals: filteredGoals,
        strengths: filteredStrengths,
        lastUpdated: new Date().toISOString(),
      };

      const userRef = doc(db, "users", auth.currentUser.uid);

      await runTransaction(db, async (transaction) => {
        const userDoc = await transaction.get(userRef);

        if (!userDoc.exists()) {
          transaction.set(userRef, {
            motivation: motivationData,
            sessionsCompleted: 1,
            exercisesDone: 0,
            exerciseHistory: [],
            routine: {}
          });
        } else {
          const currentData = userDoc.data();
          transaction.update(userRef, {
            motivation: motivationData,
            sessionsCompleted: (currentData.sessionsCompleted || 0) + 1,
          });
        }
      });

      setSavedData(motivationData);
      alert("✅ Saved Successfully!");
    } catch (error) {
      console.error("Error saving motivation:", error);
      alert("Error: " + error.message);
    } finally {
      setSaving(false);
    }
  };

  const deleteSavedItem = async (type, index) => {
    try {
      if (!auth.currentUser) return;

      const userRef = doc(db, "users", auth.currentUser.uid);
      const newData = { ...savedData };

        newData.goals = newData.goals.filter((_, i) => i !== index);
        setGoals([...newData.goals, ""]);
      } else {
        newData.strengths = newData.strengths.filter((_, i) => i !== index);
        setStrengths([...newData.strengths, ""]);
      }

      await setDoc(userRef, { motivation: newData }, { merge: true });
      setSavedData(newData);
    } catch (error) {
      console.error("Error deleting item:", error);
      alert("Error: " + error.message);
    }
  };

  if (loading) {

import { auth, db } from "../firebase/firebase";
import { doc, setDoc } from "firebase/firestore";
import { useNavigate } from "react-router-dom";

export default function Register() {
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [showPassword, setShowPassword] = useState(false);
  const [loading, setLoading] = useState(false);
  const navigate = useNavigate();

  const register = async () => {
    if (!email || !password) {
      alert("Please fill in all fields");
      return;
    }

    if (password.length < 6) {
      alert("Password must be at least 6 characters");
      return;
    }

    setLoading(true);

    try {
      console.log("Creating user account...");
      const userCred = await createUserWithEmailAndPassword(auth, email, password);
      
      console.log("User created:", userCred.user.uid);
      console.log("Creating user document in Firestore...");

      // Create user document in Firestore
      await setDoc(doc(db, "users", userCred.user.uid), {
        email: email,
        createdAt: new Date().toISOString(),
        sessionsCompleted: 0,
        exercisesDone: 0,
        routineStreak: 0,
        exerciseHistory: [],

        motivation: {
          goals: [],
          strengths: [],
        },

      console.log("User document created successfully");

      console.error("Error code:", error.code);
      console.error("Error message:", error.message);
      
      if (error.code === "auth/email-already-in-use") {
        alert("This email is already registered");
      } else if (error.code === "auth/invalid-email") {
        alert("Invalid email address");
      } else if (error.code === "auth/weak-password") {
        alert("Password is too weak");

        alert("Error: " + error.message);
      }
    } finally {
      setLoading(false);
    }

import { useState, useEffect } from "react";
import Navbar from "../components/Navbar";
import { auth, db } from "../firebase/firebase";
import { doc, setDoc, getDoc } from "firebase/firestore";

export default function Routine() {
  const [day, setDay] = useState("Monday");
  const [timeSlot, setTimeSlot] = useState("05:00");
  const [activity, setActivity] = useState("");
  const [routineData, setRoutineData] = useState({});
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);

  const days = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"];
  const timeSlots = [
    "05:00", "06:00", "07:00", "08:00", "09:00", "10:00", "11:00",
    "12:00", "13:00", "14:00", "15:00", "16:00", "17:00", "18:00",
    "19:00", "20:00", "21:00", "22:00", "23:00", "00:00"
  ];

  useEffect(() => {
    loadRoutine();
  }, []);

  const loadRoutine = async () => {
    try {
      if (!auth.currentUser) {
        console.log("No user logged in");
        setLoading(false);
        return;
      }

      console.log("Loading routine for user:", auth.currentUser.uid);
      const userRef = doc(db, "users", auth.currentUser.uid);
      const docSnap = await getDoc(userRef);

      if (docSnap.exists()) {
        console.log("Document exists:", docSnap.data());
        const data = docSnap.data();
        if (data.routine) {
          setRoutineData(data.routine);
        } else {
          console.log("No routine field in document");
        }
      } else {
        console.log("Document doesn't exist, will create on first save");
      }
    } catch (error) {
      console.error("Error loading routine:", error);
      console.error("Error code:", error.code);
      console.error("Error message:", error.message);
      // Don't show alert, just log the error
    } finally {
      setLoading(false);
    }
  };

  const saveRoutine = async () => {
    if (!activity.trim()) {
      alert("Please enter an activity");
      return;
    }

    setSaving(true);

    try {
      if (!auth.currentUser) {
        alert("Please log in again");
        setSaving(false);
        return;
      }

      const key = `${day}-${timeSlot}`;
      const userRef = doc(db, "users", auth.currentUser.uid);

      const updatedRoutine = {
        ...routineData,
        [key]: {
          day,
          time: timeSlot,
          activity,
          status: "pending",
        },
      };

      console.log("Saving routine:", updatedRoutine);

      // Use setDoc with merge to create or update
      await setDoc(
        userRef,
        {
          routine: updatedRoutine,
        },
        { merge: true }
      );

      console.log("Routine saved successfully");
      setRoutineData(updatedRoutine);
      setActivity("");
      alert("✅ Routine Saved!");
    } catch (error) {
      console.error("Error saving routine:", error);
      console.error("Error code:", error.code);
      console.error("Error message:", error.message);
      alert("Error saving: " + error.message);
    } finally {
      setSaving(false);
    }
  };

  const updateStatus = async (key, status) => {
    try {
      if (!auth.currentUser) return;

      const userRef = doc(db, "users", auth.currentUser.uid);

      const updatedRoutine = {
        ...routineData,
        [key]: {
          ...routineData[key],
          status,
        },
      };

      await setDoc(userRef, { routine: updatedRoutine }, { merge: true });
      setRoutineData(updatedRoutine);
    } catch (error) {
      console.error("Error updating status:", error);
      alert("Error: " + error.message);
    }
  };

  const deleteActivity = async (key) => {
    if (!window.confirm("Delete this activity?")) return;

    try {
      if (!auth.currentUser) return;

      const userRef = doc(db, "users", auth.currentUser.uid);
      const updatedRoutine = { ...routineData };
      delete updatedRoutine[key];

      await setDoc(userRef, { routine: updatedRoutine }, { merge: true });
      setRoutineData(updatedRoutine);
      alert("✅ Deleted");
    } catch (error) {
      console.error("Error deleting:", error);
      alert("Error: " + error.message);
    }
  };

  const getCellData = (day, time) => {
    const key = `${day}-${time}`;
    return routineData[key];
  };

  const getStatusColor = (status) => {
    if (status === "done") return "bg-green-200";
    if (status === "not-done") return "bg-red-200";
    return "bg-white";
  };

  if (loading) {
    return (
      <>
        <Navbar />
        <div className="flex justify-center items-center h-64">
          <p className="text-xl">Loading routine...</p>
        </div>
      </>
    );
  }

  return (
    <>
      <Navbar />

      <div className="p-6">
        <h2 className="text-2xl mb-4 font-bold">Routine Builder</h2>

        <div className="grid grid-cols-4 gap-2 mb-6 bg-gray-50 p-4 rounded-lg">
          <select
            className="border p-2 rounded"
            value={day}
            onChange={(e) => setDay(e.target.value)}
          >
            {days.map((d) => (
              <option key={d}>{d}</option>
            ))}
          </select>

          <select
            className="border p-2 rounded"
            value={timeSlot}
            onChange={(e) => setTimeSlot(e.target.value)}
          >
            {timeSlots.map((t) => (
              <option key={t} value={t}>
                {t}
              </option>
            ))}
          </select>

          <input
            className="border p-2 rounded"
            placeholder="Activity (e.g., Morning Run)"
            value={activity}
            onChange={(e) => setActivity(e.target.value)}
          />

          <button
            onClick={saveRoutine}
            className="bg-black text-white px-4 py-2 rounded hover:bg-gray-800 disabled:bg-gray-400"
            disabled={saving}
          >
            {saving ? "Saving..." : "Save"}
          </button>
        </div>

        <div className="overflow-x-auto border rounded-lg">
          <table className="border-collapse w-full text-sm">
            <thead>
              <tr className="bg-gray-200">
                <th className="border p-2 sticky left-0 bg-gray-200 z-10">Day / Time</th>
                {timeSlots.map((time) => (
                  <th key={time} className="border p-2 min-w-[140px]">
                    {time}
                  </th>
                ))}
              </tr>
            </thead>

            <tbody>
              {days.map((day) => (
                <tr key={day}>
                  <td className="border p-2 font-semibold sticky left-0 bg-gray-100 z-10">
                    {day}
                  </td>
                  {timeSlots.map((time) => {
                    const cellData = getCellData(day, time);
                    const key = `${day}-${time}`;

                    return (
                      <td
                        key={time}
                        className={`border p-2 ${
                          cellData ? getStatusColor(cellData.status) : ""
                        }`}
                      >
                        {cellData ? (
                          <div className="space-y-1">
                            <div className="text-xs font-medium break-words">
                              {cellData.activity}
                            </div>
                            <div className="flex gap-1 flex-wrap">
                              <button

                                className="text-xs bg-green-600 text-white px-2 py-1 rounded hover:bg-green-700"
                                title="Mark Done"
                              >
                                ✓
                              </button>
                              <button
                                onClick={() => updateStatus(key, "not-done")}
                                className="text-xs bg-red-600 text-white px-2 py-1 rounded hover:bg-red-700"
                                title="Mark Not Done"
                              >
                                ✗
                              </button>
                              <button
                                onClick={() => deleteActivity(key)}
                                className="text-xs bg-gray-600 text-white px-2 py-1 rounded hover:bg-gray-700"
                                title="Delete"
                              >
                                🗑
                              </button>
                            </div>
                          </div>
                        ) : (
                          <span className="text-gray-300">-</span>
                        )}
                      </td>
                    );
                  })}
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    </>
  );
}

import { useEffect, useState } from "react";
import Navbar from "../components/Navbar";
import { auth, db } from "../firebase/firebase";
import { doc, getDoc } from "firebase/firestore";

export default function Summary() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchData = async () => {
      try {
        const docRef = doc(db, "users", auth.currentUser.uid);
        const docSnap = await getDoc(docRef);

        if (docSnap.exists()) {
          setData(docSnap.data());
        } else {
          setData({
            sessionsCompleted: 0,
            exercisesDone: 0,
            exerciseHistory: [],
            routine: {},
          });
        }
      } catch (error) {
        console.error("Error fetching data:", error);
      } finally {
        setLoading(false);
      }
    };

    fetchData();
  }, []);

  // Calculate streak
  const calculateStreak = () => {
    if (!data?.exerciseHistory || data.exerciseHistory.length === 0) return 0;

    const sortedDates = [...new Set(data.exerciseHistory.map((ex) => ex.date))]
      .sort()
      .reverse();

    let streak = 0;
    const today = new Date().toLocaleDateString();
    const yesterday = new Date(Date.now() - 86400000).toLocaleDateString();

    // Check if exercised today or yesterday
    if (sortedDates[0] !== today && sortedDates[0] !== yesterday) {
      return 0;
    }

    let currentDate = new Date();
    for (let i = 0; i < sortedDates.length; i++) {
      const checkDate = new Date(currentDate).toLocaleDateString();

      if (sortedDates.includes(checkDate)) {
        streak++;
        currentDate = new Date(currentDate.getTime() - 86400000);
      } else {
        break;
      }
    }

    return streak;
  };

  // Calculate routine completion
  const calculateRoutineStats = () => {
    if (!data?.routine) return { done: 0, notDone: 0, pending: 0 };

    const stats = { done: 0, notDone: 0, pending: 0 };

    Object.values(data.routine).forEach((item) => {
      if (item.status === "done") stats.done++;
      else if (item.status === "not-done") stats.notDone++;
      else stats.pending++;
    });

    return stats;
  };

  // Get last 30 days exercise data for heatmap
  const getLast30Days = () => {
    const days = [];
    for (let i = 29; i >= 0; i--) {
      const date = new Date(Date.now() - i * 86400000);
      const dateStr = date.toLocaleDateString();
      const count = data?.exerciseHistory?.filter((ex) => ex.date === dateStr).length || 0;
      days.push({ date: dateStr, count });
    }
    return days;
  };

  if (loading) {
    return (
      <>
        <Navbar />
        <div className="flex justify-center items-center h-screen">
          <p className="text-xl">Loading...</p>
        </div>
      </>
    );
  }

  const streak = calculateStreak();
  const routineStats = calculateRoutineStats();
  const last30Days = getLast30Days();

  return (
    <>
      <Navbar />

      <div className="p-6 space-y-6 max-w-6xl mx-auto">
        <h2 className="text-3xl font-bold">Activity Summary</h2>

        {/* Stats Cards */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
          <div className="border rounded-lg p-6 bg-gradient-to-br from-blue-50 to-blue-100">
            <div className="text-4xl mb-2">📊</div>
            <div className="text-3xl font-bold">{data.sessionsCompleted || 0}</div>
            <div className="text-gray-600">Total Sessions</div>
          </div>

          <div className="border rounded-lg p-6 bg-gradient-to-br from-green-50 to-green-100">
            <div className="text-4xl mb-2">💪</div>
            <div className="text-3xl font-bold">{data.exercisesDone || 0}</div>
            <div className="text-gray-600">Exercises Done</div>
          </div>

          <div className="border rounded-lg p-6 bg-gradient-to-br from-orange-50 to-orange-100">
            <div className="text-4xl mb-2">🔥</div>
            <div className="text-3xl font-bold">{streak}</div>
            <div className="text-gray-600">Day Streak</div>
          </div>
        </div>

        {/* Routine Stats */}
        <div className="border rounded-lg p-6">
          <h3 className="text-xl font-semibold mb-4">Routine Completion</h3>
          <div className="grid grid-cols-3 gap-4">
            <div className="text-center p-4 bg-green-100 rounded">
              <div className="text-2xl font-bold text-green-700">
                {routineStats.done}
              </div>
              <div className="text-sm text-gray-600">Completed</div>
            </div>
            <div className="text-center p-4 bg-red-100 rounded">
              <div className="text-2xl font-bold text-red-700">
                {routineStats.notDone}
              </div>
              <div className="text-sm text-gray-600">Not Done</div>
            </div>
            <div className="text-center p-4 bg-gray-100 rounded">
              <div className="text-2xl font-bold text-gray-700">
                {routineStats.pending}
              </div>
              <div className="text-sm text-gray-600">Pending</div>
            </div>
          </div>
        </div>

        {/* Exercise Streak Heatmap */}
        <div className="border rounded-lg p-6">
          <h3 className="text-xl font-semibold mb-4">Last 30 Days Activity</h3>
          <div className="grid grid-cols-10 gap-2">
            {last30Days.map((day, index) => (
              <div
                key={index}
                className={`aspect-square rounded ${
                  day.count === 0
                    ? "bg-gray-200"
                    : day.count === 1
                    ? "bg-green-300"
                    : day.count === 2
                    ? "bg-green-500"
                    : "bg-green-700"
                }`}
                title={`${day.date}: ${day.count} exercises`}
              ></div>
            ))}
          </div>
          <div className="flex justify-end gap-2 mt-4 text-xs text-gray-600">
            <span>Less</span>
            <div className="w-4 h-4 bg-gray-200 rounded"></div>
            <div className="w-4 h-4 bg-green-300 rounded"></div>
            <div className="w-4 h-4 bg-green-500 rounded"></div>
            <div className="w-4 h-4 bg-green-700 rounded"></div>
            <span>More</span>
          </div>
        </div>

        {/* Recent Exercise History */}
        {data?.exerciseHistory && data.exerciseHistory.length > 0 && (
          <div className="border rounded-lg p-6">
            <h3 className="text-xl font-semibold mb-4">Recent Exercises</h3>
            <div className="space-y-2">
              {data.exerciseHistory
                .slice(-10)
                .reverse()
                .map((ex, index) => (
                  <div
                    key={index}
                    className="flex justify-between items-center border-b pb-2"
                  >
                    <span className="font-medium">{ex.exerciseName}</span>
                    <span className="text-sm text-gray-600">
                      {new Date(ex.timestamp).toLocaleString()}
                    </span>
                  </div>
                ))}
            </div>
          </div>
        )}
      </div>
    </>
  );

export const sendMessageToAI = async (message) => {
  try {
    const model = genAI.getGenerativeModel({
      model: "gemini-1.5-flash",
    });

    const prompt = `You are an AI mental wellbeing assistant for athletes.
Give short supportive responses and suggest relaxation,
focus, or confidence exercises when appropriate.

User: ${message}
`;

    const result = await model.generateContent(prompt);
    const response = result.response;
    const text = response.text();

    return text;
